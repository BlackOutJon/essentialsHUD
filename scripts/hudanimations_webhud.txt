event HudHealthBonusPulse
{
	Animate RedPlayerHealth						FgColor		"HealthBuff"	Linear 0.0 0.0
	Animate BluePlayerHealth					FgColor		"HealthBuff"	Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpec			FgColor		"HealthBuff"	Linear 0.0 0.0
	Animate PlayerStatusHealthBonusImage    	Alpha       "0"       		Linear 0.0 0.0
	RunEvent HudHealthBonusPulseLoop			0.4
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse				0.0
	StopEvent HudHealthBonusPulseLoop			0.0
	
	Animate RedPlayerHealth						FgColor		"HudWhite"		Linear 0.0 0.0
	Animate BluePlayerHealth					FgColor		"HudWhite"		Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpec			FgColor		"HudWhite"		Linear 0.0 0.0
	Animate PlayerStatusHealthBonusImage    	Alpha       "255"       	Linear 0.0 0.0
}
